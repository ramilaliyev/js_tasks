#Доп: Фильтрация массива

Создайте функцию которая в качестве аргумента принимает целое число, и проверяет четное оно или нет. Затем создайте произвольный массив целых чисел. И используя функцию проверки числа на четность, создайте из готового массива новый, к оторый будет содержать только четные числа. Результат выведите в консоль. 

        const mixedArray = [3, 13, 74, 14, 66, 15, 22, 4];

        const isEven = num => { 
            if (num % 2 == 0) {
                return true;
            }
        };

        var filterArray = (arr, f) => {
            var fArr = [];
            
            for (var i = 0; i < arr.length; i++) {
                if (f(arr[i])) {
                    fArr.push(arr[i]);
                } 
            }

            return fArr;
        }

        console.log(filterArray(mixedArray, isEven));
